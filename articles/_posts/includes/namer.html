import os
import re
def naturalSort(l):
    l.sort(key=lambda x:[(int(c) if c.isdigit() else c) for c in re.split('([0-9]+)', x)])
    return l
i=1
dirname='home'
for img in naturalSort([ x for x in os.listdir(dirname) if x.endswith('.png')]):
    name=f'{i:03}.png'
    print(f'{img}->{name}')
    os.rename(os.path.join(dirname,img),os.path.join(dirname,name))
    i+=1
